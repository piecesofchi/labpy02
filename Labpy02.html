program membandingkan 3 input bilangan, apabila penjumlahan 2 bilangan hasilnya sama dengan bilangan lainnya, maka cetak pernyataan “BENAR”

a = int(input("Masukkan bilangan A: ")) b = int(input("Masukkan bilangan B: ")) c = int(input("Masukkan bilangan C: ")) if a+b == c or b+c == a or c+a == b: print("BENAR") else: print("SALAH")

Program Pemesanan Tiket Bioskop

# Harga tiket harga_tiket_reguler = 50000 harga_tiket_vip = 100000 # Input dari user tipe_tiket = input("Masukkan tipe tiket (reguler/vip): ").lower() member = input("Apakah Anda memiliki kartu member? (ya/tidak): ").lower() # Menghitung harga berdasarkan tipe tiket if tipe_tiket == "reguler": harga = harga_tiket_reguler elif tipe_tiket == "vip": harga = harga_tiket_vip else: print("Tipe tiket tidak valid.") exit() # Cek status member untuk memberikan diskon if member == "ya": harga = harga * 0.8 # Diskon 20% # Menampilkan total harga print(f"Total harga yang harus dibayar: Rp{int(harga)}")
